105 private void esporta() {
106 if(chooser.showSaveDialog(this) == JFileChooser.APPROVE_OPTION) {
107 String nomefile = chooser.getSelectedFile().getAbsolutePath() + "." + cbFormato.getSelectedItem().toString().toLowerCase();
108
109 if(cbFormato.getSelectedItem().equals("CSV")) {
110 exporter = new EsportatoreCSV(nomefile, panIniz.getElencoVoci());
111 exporter.esporta();
112 } else if(cbFormato.getSelectedItem().equals("txt")) {
113 exporter = new EsportatoreTxt(nomefile, panIniz.getElencoVoci());
114 exporter.esporta();
115 }
116
117 JOptionPane.showMessageDialog(panIniz.getFrame(), "Bilancio esportato in " + nomefile, "Esportazione", JOptionPane.INFORMATION_MESSAGE);
118 panIniz.getFrame().cambiaScheda("PannelloIniziale");
119 }
